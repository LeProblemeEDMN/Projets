#version 150

in vec2 position;

out vec2 blurTexturesCoords[11];
uniform float targetWidth;

void main(void){

	gl_Position = vec4(position, 0.0, 1.0);
	vec2 centerTexCoords = position * 0.5 + 0.5;

	float pixelSize=1.0/targetWidth;
	
	
	for(int x=-5;x<=5;x++){
		blurTexturesCoords[x+5]=centerTexCoords+vec2(pixelSize*x,0.0);
	}
}
