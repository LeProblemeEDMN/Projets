#version 400 core

in vec3 position;

out vec3 toCameraVector;

uniform mat4 transformationMatrix;
uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;


void main(void){
	
	vec4 worldPosition=transformationMatrix*vec4(position,1.0);

	vec4 positionRelativToCamera=viewMatrix*worldPosition;
	gl_Position =projectionMatrix* positionRelativToCamera;
	//gl_Position =vec4(position,1.0);
	//toCameraVector=(gl_Position.xyz/gl_Position.w)*0.5+0.5;
	toCameraVector.xyz=positionRelativToCamera.xyz;
}
